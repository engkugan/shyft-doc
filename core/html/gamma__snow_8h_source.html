175  double corr_lwc(const double z1, const double a1, const double b1,
176  double z2, const double a2, const double b2) const
177  {
178  double zmax = z1;
179  double zmin = 0.0;
180  double dz = fabs(zmax - zmin);
181  double dz_old = dz;
182  const double rtol = 1.0e-4;
183 
184  const double Q1 = calc_q(a1, b1, z1);
185  const double precision = Q1*rtol;
186  double Q2 = calc_q(a2, b2, z2);
187  double f = Q2 - Q1;
188 
189  double df = calc_df(a2, b2, z2);
190 
191  size_t i;
192  size_t max_iter = 30;
193  for (i=0; i < max_iter; ++i) {
194  if (((z2 - zmax)*df - f)*((z2 - zmin)*df - f) > 0.0 || fabs(2.0*f) > fabs(dz_old*df)) {
195  dz_old = dz;
196  dz = (zmax - zmin)*0.5;
197  z2 = zmin + dz;
198  } else {
199  dz_old = dz;
200  dz = f/df;
201  z2 -= dz;
202  }
203  if (fabs(dz) < precision)
204  break;
205  // Prepare next iteration
206  Q2 = calc_q(a2, b2, z2);
207  f = Q2 - Q1;
208  df = calc_df(a2, b2, z2);
209  if (f < 0.0) zmin = z2;
210  else zmax = z2;
211  }
212  #ifdef __UNIT_TEST__
213  #if VERBOSE>0
214  std::cout << "Num iter in corr_lwc = " << i << std::endl;
215  #endif
216  #endif
217  if (i < max_iter) return z2;
218  else throw std::runtime_error("corr_lwc did not converge"); // No convergence in max_iter iterations!
219  }

221  void calc_snow_state(const double shape, const double scale, const double y0, const double lambda, const double lwd,
222  const double max_water_frac, const double temp_swe, double& swe, double& sca) const
223  {
224  double y = 0.0;
225  double y1 = 0.0;
226  const double m = shape*scale;
227  if (lambda <= 0.0) { // accumulated melt is -1.
228  swe = m;
229  sca = 1.0 - y0;
230  } else if (lambda/scale > 1.3*shape + 20.0) {
231  swe = sca = 0.0;
232  return;
233  } else {
234  const double x = lambda/scale;
235  y = gamma_p(shape, x);
236  y1 = y - exp(shape*log(x) - x - lgamma(shape))/shape;
237  swe = m*(1.0 - y1) - lambda*(1 - y);
238  sca = (1.0 - y)*(1.0 - y0);
239  }
240  if (lwd > m) swe *= 1.0 + max_water_frac;
241  else if (lwd > 0.0) {
242  const double sat = lwd/max_water_frac;
243  const double x = sat/scale;
244  const double ssa = gamma_p(shape, x);
245  const double ssa1 = ssa - exp(shape*log(x) - x - lgamma(shape))/shape;
246  const double liqwat = max_water_frac*(m*(ssa1 - y1) + sat*(1.0 - ssa) - lambda*(1.0 - y));
247  swe += liqwat;
248  }
249  swe += temp_swe;
250  swe *= 1.0 - y0;
251  }
