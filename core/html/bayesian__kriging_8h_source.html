273  arma::mat::fixed<2,1> E_beta_pri, E_beta_w_pri, E_beta_post, beta_hat;
274  // These matrices sizes vary with the number valid sources and the number of destinations.
275  arma::mat K, k, F, f, K_inv, E_beta_w, omega, BM, T_obs, E_temp_post;
276 
277  // Prior data
278  E_beta_pri.at(0, 0) = 0.0; // Old code says this is ok. TODO: Check assumption.
279  arma::mat22 eye22 = arma::diagmat(arma::vec(2, arma::fill::ones));
280 
281  // Gather spatial data for all stations and destinations
282  utils::build_elevation_matrices(source_begin, source_end, destination_begin, destination_end, F, f);
283  utils::build_covariance_matrices(source_begin, source_end, destination_begin, destination_end, parameter, K, k);
284 
285  // Build full operators
286 
287  K_inv = K.i();
288  H_inv.operator=(F.t()*K_inv*F);
289  if (arma::rank(H_inv) == 1) {
290  throw std::runtime_error("The bayestian temperature kriging algorithm needs at least two sources at different heights.");
291  }
292  H.operator=(H_inv.i());
293  G_inv = H_inv;
294  G_inv.at(1, 1) += 1/(parameter.temperature_gradient_sd()*parameter.temperature_gradient_sd());
295  G.operator=(G_inv.i());
296  GH_inv.operator=(G*H_inv);
297  BM = (f - F.t()*K_inv*k).t()*(eye22 - GH_inv);
298  E_beta_w = H*F.t()*K_inv; // beta_est_weights
299  omega = k.t()*K_inv; // krig_weights
300  // Reduced matrices
301  arma::mat F_r, E_beta_w_r, omega_r, GH_inv_r, BM_r;
302 
303  // Matrix pointers used in the time loop
304  arma::mat *F_p, *E_beta_w_p, *omega_p, *GH_inv_p, *BM_p;
305 
306  const size_t num_sources = std::distance(source_begin, source_end);
307  std::vector<TSA> source_accessors;
308  source_accessors.reserve(num_sources);
309  std::for_each(source_begin, source_end, [&] (const typename S::value_type& source)
310  { source_accessors.emplace_back(TSA(source.temperatures(), time_axis)); });
311  // Time step loop
312  const size_t num_timesteps = time_axis.size();
313  std::vector<double> temperatures;
314  temperatures.reserve(num_sources);
315  size_t idx;
316 
317  std::vector<arma::uword> valid_inds, prev_valid_inds;
318  valid_inds.reserve(num_sources);
319  for (size_t t_step=0; t_step < num_timesteps; ++t_step) {
320  temperatures.clear();
321  prev_valid_inds = valid_inds;
322  valid_inds.clear();
323  idx = 0;
324  for (auto& a: source_accessors) {
325  double v = a.value(t_step);
326  if (std::isfinite(v)) {
327  valid_inds.push_back((arma::uword)idx);
328  temperatures.push_back(v);
329  }
330  ++idx;
331  }
332  if (valid_inds != prev_valid_inds) {
333  if (valid_inds.size() == 0) {
334  std::cout << "period("<< t_step <<"| " << num_timesteps << ") = " << time_axis(t_step) << std::endl;
335  throw std::runtime_error("No valid sources for time period, giving up.");
336  }
337  if (valid_inds.size() == num_sources) {
338  // Use full operators
339  F_p = &F;
340  E_beta_w_p = &E_beta_w;
341  omega_p = &omega;
342  GH_inv_p = &GH_inv;
343  BM_p = &BM;
344  } else {
345  // Build new reduced operators from full operators
346  arma::uvec sub_idx(valid_inds);
347  F_r = F.rows(sub_idx);
348  K_inv = K.submat(sub_idx, sub_idx).i();
349  H_inv.operator=(F_r.t()*K_inv*F_r);
350  H.operator=(H_inv.i());
351  G_inv = H_inv;
352  G_inv.at(1, 1) += 1/(parameter.temperature_gradient_sd()*parameter.temperature_gradient_sd());
353  G.operator=(G_inv.i());
354  GH_inv_r = G*H_inv;
355  arma::mat k_red = k.rows(sub_idx);
356  BM_r = (f - F_r.t()*K_inv*k_red).t()*(eye22 - GH_inv_r);
357  E_beta_w_r = H*F_r.t()*K_inv; // beta_est_weights
358  omega_r = k_red.t()*K_inv; // krieg_weights
359  // Assign pointers
360  F_p = &F_r;
361  E_beta_w_p = &E_beta_w_r;
362  omega_p = &omega_r;
363  GH_inv_p = &GH_inv_r;
364  BM_p = &BM_r;
365  }
366  }
367 
368  // Build prior data for time step:
369  E_beta_pri.at(1, 0) = parameter.temperature_gradient(time_axis(t_step));
370  E_beta_w_pri.operator=((eye22 - *GH_inv_p)*E_beta_pri);
371 
372  // Fill T_obs with valid temperatures
373  T_obs.set_size((arma::uword)valid_inds.size(), 1);
374  std::copy(std::begin(temperatures), std::end(temperatures), T_obs.begin_col(0));
375  // Core computational work here:
376  beta_hat.operator=((*E_beta_w_p)*T_obs);
377  arma::mat T_hat = f.t()*beta_hat + (*omega_p)*(T_obs - (*F_p)*beta_hat);
378  E_beta_post.operator=((*GH_inv_p)*beta_hat + E_beta_w_pri);
379  E_temp_post = arma::vec(T_hat - (*BM_p)*(beta_hat - E_beta_pri));
380 
381  arma::uword dist = 0;
382  for (D d=destination_begin; d != destination_end; ++d)
383  d->set_temperature(t_step, E_temp_post.at(dist++));
384 
385  }
