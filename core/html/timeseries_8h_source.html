310  size_t hint_based_search(const S& source,const utcperiod& p, size_t i) {
311  if (source.size() == 0)
312  return std::string::npos;
313  const size_t n=source.size();
314  if (i != std::string::npos && i <n ) { // hint-based search logic goes here:
315  const size_t max_directional_search=5;// +-5 just a guess for what is a reasonable try upward/downward
316  auto ti = source.get(i).t;
317  if (ti == p.start) {// direct hit and extreme luck ?
318  ;// just use it !
319  } else if(ti < p.start) { // do a local search upward to see if we find the spot we search for
320  size_t j=0;
321  while(ti < p.start && ++j <max_directional_search && i<n) {
322  ti=source.get(i++).t;
323  }
324  if(ti>=p.start || i==n ) // we startet below p.start, so we got one to far(or at end), so revert back one step
325  --i;
326  else //if( ti !=p.start) // bad luck, local bounded search fail,
327  i=source.index_of(p.start);// cost passed to binary search
328  } else if(ti>p.start) { // do a local search downwards from last index, maybe we are lucky
329  size_t j=0;
330  while(ti>p.start && ++j <max_directional_search && i >0) {
331  ti=source.get(--i).t;
332  }
333  if(ti>p.start && i>0) // if we are still not before p.start, and i is >0, there is a hope to find better index, otherwise we are at/before start
334  i=source.index_of(p.start); // bad luck searching downward, need to use binary search.
335  }
336  } else // no hint given, just use binary search to establish the start.
337  i = source.index_of(p.start);
338  return i;
339  }

361  double average_value(const S& source, const utcperiod& p, size_t& last_idx,bool linear=true) {
362  const size_t n=source.size();
363  if (n == 0) // early exit if possible
364  return shyft::nan;
365  size_t i=hint_based_search(source,p,last_idx); // use last_idx as hint, allowing sequential periodic average to execute at high speed(no binary searches)
366 
367  if(i==std::string::npos) // this might be a case
368  return shyft::nan; // and is equivalent to no points, or all points after requested period.
369 
370  point l;// Left point
371  bool l_finite=false;
372 
373  double area = 0.0; // Integrated area over the non-nan parts of the time-axis
374  utctimespan tsum=0; // length of non-nan f(t) time-axis
375 
376  while(true) { //there are two exit criteria: no more points, or we pass the period end.
377  if(!l_finite) {//search for 'point' anchor phase
378  l=source.get(i++);
379  l_finite=std::isfinite(l.v);
380  if(i==n) { // exit condition
381  if(l_finite && l.t < p.end ) {//give contribution
382  utctimespan dt= p.end-l.t;
383  tsum += dt;
384  area += dt* l.v; // extrapolate value flat
385  }
386  break;//done
387  }
388  if (l.t >= p.end) {//also exit condition, if the point time is after period we search, then no anchor to be found
389  break;//done
390  }
391  } else { // got point anchor l, search for right point/end
392  point r=source.get(i++);// r is guaranteed > p.start due to hint-based search
393  bool r_finite=std::isfinite(r.v);
394  utcperiod px(std::max(l.t,p.start),std::min(r.t,p.end));
395  utctimespan dt= px.timespan();
396  tsum += dt;
397 
398  // now add area contribution for l..r
399  if(linear && r_finite) {
400  double a= (r.v-l.v)/(r.t-l.t);
401  double b= r.v - a*r.t;
402  //area += dt* 0.5*( a*px.start+b + a*px.end+b);
403  area += dt * ( 0.5*a*(px.start+px.end)+ b);
404  } else { // flat contribution from l max(l.t,p.start) until max time of r.t|p.end
405  area += l.v*dt;
406  }
407  if(i==n) { // exit condition: final value in sequence, then we need to finish, but
408  if(r_finite && r.t < p.end) {// add area contribution from r and out to the end of p
409  dt= p.end-r.t;
410  tsum += dt;
411  area += dt* r.v; // extrapolate value flat
412  }
413  break;//done
414  }
415  if(r.t >= p.end)
416  break;//also exit condition, done
417  l_finite=r_finite;
418  l=r;
419  }
420 
421  }
422  last_idx=i-1;
423  return tsum?area/tsum:shyft::nan;
424  }

657  double kling_gupta_goal_function(const TSA1& observed_ts, const TSA2& model_ts,double s_r,double s_a,double s_b) {

659  for (size_t i = 0; i < observed_ts.size(); ++i) {
660  double tv = observed_ts.value(i);

663  rs.add(tv, dv);
664  }

667  double us = rs.stddev_y();
668  double uo = rs.stddev_x();
669  double r = rs.correlation();
670  double a = qs / qo;
671  double b = us / uo;
672  if(!isfinite(a)) a = 1.0;//could happen in theory if qo is zero
673  if(!isfinite(b)) b = 1.0;// could happen if uo is zero
674  // We use EDs to scale, and KGEs = (1-EDs) with max at 1.0, we use 1-KGEs to get minimum as 0 for minbobyqa
675  return /*EDs=*/ sqrt(std::pow(s_r*(r - 1), 2) + std::pow(s_a*(a - 1), 2) + std::pow(s_b*(b - 1), 2));
