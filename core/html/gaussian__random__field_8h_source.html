48  inline arma::mat::fixed<2, 2> anisotropy_mapping(double anisotropy_ratio, double anisotropy_direction) {
49  arma::mat::fixed<2, 2> T;
50 
51  if (anisotropy_ratio < 1.0)
52  throw semivariogram_error();
53 
54  if (anisotropy_ratio > 1.0 + 1.0e-10) {
55  const double kernel = (90.0 - anisotropy_direction)*M_PI/180.0;
56  T.at(0, 0) = cos(kernel);
57  T.at(0, 1) = sin(kernel);
58  T.at(1, 0) = -sin(kernel)*anisotropy_ratio;
59  T.at(1, 1) = cos(kernel)*anisotropy_ratio;
60  } else
61  T.operator=( arma::eye(2, 2));
62  return T;
63  }

286  void calculate_gibbs_weights(const arma::mat& cov,
287  const std::vector<unsigned int>& sub_inds,
288  arma::mat& krig_weights) {
289  const size_t m = cov.n_rows - 1; // Matrix dimension for krig LA problem
290  krig_weights.set_size(sub_inds.size(), m + 1);
291  arma::vec x(m);
292  arma::vec weights(m + 1);
293  arma::uvec indices(cov.n_rows);
294  for (unsigned int i = 0; i < cov.n_rows; ++i) indices.at(i) = i;
295  for (size_t idx = 0; idx < sub_inds.size(); ++idx) {
296  const unsigned int zero_index = sub_inds[idx];
297  // Build submatrix index vector by excluding zero_index station
298  std::vector<unsigned int> sub_vec(indices.begin(), indices.begin() + zero_index);
299  sub_vec.insert(sub_vec.end(), indices.begin() + zero_index + 1, indices.end());
300  const arma::uvec sub_indices(sub_vec);
301  const arma::uvec::fixed<1> col = {zero_index};
302 
303  // Solve simple krig problem
304  arma::solve(x, cov(sub_indices, sub_indices), cov(sub_indices, col));
305 
306  // Insert 0.0 as weight at (zero_index, zero_index) to preserve index mapping
307  // This allows efficient matrix-vector arithmetics when calculating the gibbs estimates
308  if (zero_index > 0)
309  weights.subvec(0, zero_index - 1) = x.subvec(0, zero_index - 1);
310  weights.at(zero_index) = 0.0;
311  if (zero_index < m)
312  weights.subvec(zero_index + 1, m) = x.subvec(zero_index, m - 1);
313  krig_weights.row(idx) = weights.t();
314  }
315  }

322  void calculate_gibbs_estimates(const std::vector<unsigned int>& sub_inds, // Sub indices to calculate over
323  const std::vector<double>& data, // Precipitations
324  const arma::mat& weights, // Kriging weights
325  const arma::mat& cov, // Full covariance matrix
326  const double sill,
327  std::vector<double>& mu, // Kriging estimate
328  std::vector<double>& var) { // Kriging variance
329  typedef arma::conv_to<std::vector<double>> to_std_vec;
330  typedef arma::conv_to<arma::vec> to_arma_vec;
331  typedef arma::conv_to<arma::uvec> to_arma_uvec;
332 
333  const double obs_mean = arma::mean(to_arma_vec::from(data));
334 
335  const arma::uvec sub_indices = to_arma_uvec::from(sub_inds);
336  mu = to_std_vec::from(
337  obs_mean*(1.0 - arma::sum(weights, 1)) + weights*to_arma_vec::from(data)
338  );
339  var.resize(sub_inds.size());
340 
341  for (size_t i = 0; i < sub_indices.n_rows; ++i) {
342  // sigma^2 = C(0) - lambda^T*k, see page 8 in http://people.ku.edu/~gbohling/cpe940/Kriging.pdf
343  const size_t idx = sub_indices.at(i);
344  var[i] = sill - to_arma_vec::from(weights.row(i)*cov.col(idx)).at(0);
345  }

367  }

372  arma::mat K(m, m);
373  arma::vec k(m);
374  K.submat(0, 0, m - 2, m - 2) = station_covariances(sub_indices, sub_indices);

376  K.col(m - 1) = arma::vec(m, arma::fill::ones);

378  k.subvec(0, m - 2) = dest.source_covariances()(sub_indices);
379  k.at(m - 1) = 1.0;

381  w = l.subvec(0, m - 2);

384  // the actual implementation of it. Figure out if solving a second kringing problem is needed, and if so, recalc

647  std::vector<TSA> source_accessors;
648  source_accessors.reserve(num_sources);
649  std::for_each(source_begin, source_end, [&] (const typename S::value_type& source)
650  { source_accessors.emplace_back(TSA(source.precipitations(), time_axis)); });
651 
652  const size_t num_timesteps = time_axis.size();
653  std::vector<size_t> valid_indices; valid_indices.reserve(num_sources);
654  std::vector<size_t> prev_valid_indices; prev_valid_indices.reserve(num_sources);
655  typedef typename S::value_type* s_p_type;
656  std::vector<s_p_type> valid_sources;
657 
658  // Variables used to store the values needed to change from gamma to normal distribution
659  // Mean, cv and p0 need to be determined based on a separate pass through the source data.
660  std::vector<double> precipitations;
661  std::vector<double> precipitations_mean;
662  std::vector<double> precipitations_cv;
663  std::vector<double> precipitations_p0;
664  precipitations.reserve(num_sources);
665  precipitations_mean.reserve(num_sources);
666  precipitations_cv.reserve(num_sources);
667  precipitations_p0.reserve(num_sources);
668 
669  for (size_t t_step = 0; t_step < num_timesteps; ++t_step) {
670  std::vector<unsigned int> pos_inds;
671  std::vector<unsigned int> zero_inds;
672  pos_inds.reserve(num_sources);
673  zero_inds.reserve(num_sources);
674  valid_sources.clear();
675  auto source_iter = source_begin;
676  for(size_t i = 0; i < num_sources; ++i) {
677  const double precipitation = source_accessors[i].value(t_step);
678  const double precipitation_mean = 0.5; // TODO: Estimate!
679  const double precipitation_cv = 0.2; // TODO: Estimate!
680  const double precipitation_p0 = 0.3; // TODO: Estimate!
681  if (std::isfinite(precipitation)) {
682  valid_indices.emplace_back(i);
683  valid_sources.emplace_back(&(*source_iter));
684  precipitations.emplace_back(precipitation);
685  precipitations_mean.emplace_back(precipitation_mean);
686  precipitations_cv.emplace_back(precipitation_cv);
687  precipitations_p0.emplace_back(precipitation_p0);
688  if (precipitation > 0.0) pos_inds.emplace_back(i);
689  else zero_inds.emplace_back(i);
690  }
691  ++source_iter;
692  }
693  std::vector<double> transformed_precipitation;
694  grf::gamma_transform(precipitations,
695  precipitations_mean,
696  precipitations_cv,
697  precipitations_p0,
698  transformed_precipitation);
699  const double positive_mean = pos_inds.size() > 0 ? grf::subset_mean(transformed_precipitation, pos_inds) : 0.0;
700  const double positive_var = pos_inds.size() > 1 ? grf::subset_var(transformed_precipitation, pos_inds) : 0.0;
701 
702  const double zero_var = 0.35; // TODO: Why?
703  const double zero_mean = -0.75; // TODO: Why?
704  const size_t n_z = zero_inds.size();
705  const size_t n_p = pos_inds.size();
706 
707  const double tmp = (positive_mean - zero_mean)/(n_p + n_z);
708  const double total_var = (positive_var*n_p + zero_var*n_z)/(n_p + n_z) + n_p*n_z*tmp*tmp;
709  const double nugget = parameter.nugget();
710 
711  arma::mat station_distances;
712  grf::calculate_anisotropy_distances(begin(valid_sources), end(valid_sources),
713  parameter.anisotropy_ratio(), parameter.anisotropy_direction(), station_distances);
714  grf::semi_variogram semi_var(nugget, total_var + nugget, parameter.range());
715 
716  typedef boost::math::policies::policy<boost::math::policies::digits10<3>> acc_policy;
717  typedef boost::math::normal_distribution<double, acc_policy> normal_dist;
718  normal_dist n_d;

735  zero_inds, max_zero_transformed_precipitation,
736  semi_var, transformed_precipitation);
737  }
738  arma::mat source_dest_dists;
739  if (valid_indices != prev_valid_indices) {
740  grf::calculate_anisotropy_distances(begin(valid_sources),
741  end(valid_sources),
742  destination_begin,
743  destination_end,
744  parameter.anisotropy_ratio(),
745  parameter.anisotropy_direction(),
746  source_dest_dists);
747  }
